import pandas as pdimport pickle as pkfrom sklearn.feature_extraction.text import TfidfVectorizerimport streamlit as st# Load the pre-trained model and vectorizermodel = pk.load(open('model.pkl', 'rb'))scaler = pk.load(open('scaler.pkl', 'rb'))# User input for movie reviewreview = st.text_input('Enter Movie Review')if st.button('Predict'):    # Vectorize and scale the input review    review_vector = scaler.transform([review]).toarray()    # Predict the sentiment    prediction_prob = model.predict_proba(review_vector)  # get prediction probabilities    prediction = model.predict(review_vector)    # Thresholds for neutral sentiment    neutral_threshold = 0.45  # This is arbitrary, you can adjust it    # Interpret the prediction result    if prediction_prob[0][1] > (1 - neutral_threshold):        sentiment = 'Positive Review'        rating = 5    elif prediction_prob[0][1] < neutral_threshold:        sentiment = 'Negative Review'        rating = 1    else:        sentiment = 'Neutral Review'        rating = 3        # Display the sentiment and rating    st.write(sentiment)    st.write(f"Rating: {rating}/5")
